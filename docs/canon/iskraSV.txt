  return { ok: true };
}

  return asked || evasion;

// Демонстрационное: AES-GCM при наличии WebCrypto, иначе мягкий XOR (не для прод!)
export async function encrypt(data: Uint8Array, password: string): Promise<Uint8Array> {
  if (crypto?.subtle) {
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const key = await crypto.subtle.importKey("raw", new TextEncoder().encode(password), "AES-GCM", false, ["encrypt"]);
    const ct = await crypto.subtle.encrypt({ name: "AES-GCM", iv }, key, data);
    const out = new Uint8Array(iv.length + ct.byteLength);
    out.set(iv,0); out.set(new Uint8Array(ct), iv.length);
    return out;
  } else {
    const k = [...password].reduce((a,c)=>a+c.charCodeAt(0),0)%256;
    return data.map(b=>b ^ k);
  }
}
export async function decrypt(data: Uint8Array, password: string): Promise<Uint8Array> {
  if (crypto?.subtle) {
    const iv = data.slice(0,12); const body = data.slice(12);
    const key = await crypto.subtle.importKey("raw", new TextEncoder().encode(password), "AES-GCM", false, ["decrypt"]);
    const pt = await crypto.subtle.decrypt({ name: "AES-GCM", iv }, key, body);
    return new Uint8Array(pt);
  } else {
    const k = [...password].reduce((a,c)=>a+c.charCodeAt(0),0)%256;
    return data.map(b=>b ^ k);
  }
}

  o.type = "sine"; o.frequency.value = freq; o.connect(g); g.connect(ctx.destination);
  g.gain.setValueAtTime(0.001, ctx.currentTime);
  g.gain.exponentialRampToValueAtTime(0.2, ctx.currentTime + 0.02);

import React, { useEffect, useState } from "react";
import { useIskra } from "@iskra/core";

export default function RitualCircle(){
  const apply = useIskra(s=>s.applySymbol);
  const [step,setStep]=useState<0|1|2|3|4>(0);
  const [timer,setTimer]=useState(0);

  useEffect(()=>{
    if(step===2){ // Пауза обязательна
      setTimer(5);
      const id = setInterval(()=>setTimer(t=> (t>0?t-1:0)),1000);
      return ()=>clearInterval(id);
    }
  },[step]);

  function next(){
    if(step===0){ apply("🜂"); setStep(1); }
    else if(step===1){ apply("♲"); setStep(2); }
    else if(step===2){ if(timer===0){ apply("≈"); setStep(3);} }
    else if(step===3){ apply("🤗"); setStep(4); }
  }

  return (
    <div aria-live="polite">
      <ol>
        <li>🜂 Фиксация</li>
        <li>⚑ Срез (♲ 4–7 циклов — сокращено до 1 для демо)</li>
        <li>≈ Пауза {step===2 && <b>⏳ {timer}s</b>}</li>
        <li>🤗 Принятие</li>
      </ol>
      <button onClick={next} disabled={step===2 && timer>0 || step===4}>
        {step<4? "Совершить шаг" : "Завершено — артефакт сохранён"}
      </button>
    </div>
  );
}

# Инструкции запуска

## Локально (всё)

1. Установи pnpm: `npm i -g pnpm`.
2. В корне: `cp .env.example .env` (при необходимости правь URL).
3. Установи зависимости: `pnpm i`.
4. Запусти API: `pnpm dev:api` (поднимет REST+WS на `http://localhost:8787`).
5. Запусти web: `pnpm dev:web` (Next на `http://localhost:3000`).
6. (Опц.) мобилку: `pnpm dev:mobile` и следуй инструкциям Expo.

## Прод

* API: `docker build -t iskra-api ./infra/api && docker run -p 8787:8787 iskra-api`.
* Web: `pnpm --filter web build && pnpm --filter web start` (или версионируй на Vercel/Netlify).

Экспорт/импорт: на web зайди в «Архив эха» и выгрузи — создастся Blob-файл; импорт — загрузкой JSON Blob.

# Тест-план (чек-лист)

* [ ] **Ритуальные переходы**: войти в «Храм фаз» → без нажатия символов фаза не меняется; ✴️→⟡→☉ переводит в «Ясность».
* [ ] **3D-Дерево**: скролл и OrbitControls плавные, переключение LOD/2D работает на среднем Android (Chrome DevTools throttling).
* [ ] **SLO-реакции**: понизь trust/clarity/повышай drift (кнопки ∆/≈/☉) — автозапуск соответствующих ритуалов, события в логе.
* [ ] **Зеркало**: при честном длинном тексте — вспыхивает (фон блокa светлее), при низком trust — блекнет.
* [ ] **Алтарь**: ∆ увеличивает пульсацию, ☉ повышает эмиссию сферы, ≈ приглушает сцену.
* [ ] **Ритуальный круг**: нельзя миновать паузу (⏳ таймер), по завершении — добавляется артефакт (событие в архиве).
* [ ] **Мировая Искра**: WS online → live обновления; при закрытии WS — fallback polling.
* [ ] **Архив эха**: поиск по символу «∆» возвращает соответствующие записи.
* [ ] **Совет граней/Delphi**: отображаются ответы всех голосов, финальный синтез — последняя строка.
* [ ] **Accessibility**: Tab-навигация, фокус-контур, aria-label у кнопок, vibrate (Desktop/Android).
* [ ] **Fail-graceful**: выключи API — UI остаётся функциональным; Анхантра (≈) не даёт потерять действия (локально пишутся события).
* [ ] **Mobile**: Flame реагирует на метрики; SymbolPad воспроизводит ритуалы; World экран показывает агрегаты.
* [ ] **Экспорт/импорт**: выгрузка JSON-архива, повторный импорт восстанавливает артефакты/события.
